// ============================================ // PROCESSADOR DE EMAILS // Configurar trigger: a cada 5 ou 10 minutos // ============================================ function processarEmails() { const remetentesPermitidos = listarRemetentes(); if (remetentesPermitidos.length === 0) return; const threads = GmailApp.search('is:unread label:inbox', 0, 20); threads.forEach(thread => { const messages = thread.getMessages(); messages.forEach(msg => { if (msg.isUnread()) { const from = msg.getFrom().match(/<(.+?)>/)?.[1] || msg.getFrom(); const fromClean = from.toLowerCase().trim(); if (!remetentesPermitidos.some(r => r.toLowerCase().trim() === fromClean)) return; const body = msg.getPlainBody(); const attachments = obterAnexos(msg); const linksDrive = extrairLinksDrive(body); if (attachments.length === 0 && linksDrive.length === 0) return; // Extrair nomes do corpo do email (separados por ; ou quebra de linha) const nomes = extrairNomes(body); if (nomes.length === 0) return; // Se múltiplos nomes, compartilham a mesma pasta const grupo = nomes.length > 1 ? Utilities.getUuid().split('-')[0] : null; let pastaGrupo = null; // Verificar se algum dos nomes já existe (usa a pasta dele) if (nomes.length > 1) { for (const nome of nomes) { const existente = buscarClientePorNome(nome); if (existente && existente.pastaDrive) { pastaGrupo = existente.pastaDrive; break; } } // Se nenhum existe, busca pasta existente ou cria nova if (!pastaGrupo) { const primeirosNomes = nomes.map(n => n.split(' ')[0]).join(' e '); const parentFolder = DriveApp.getFolderById(CONFIG.DRIVE_FOLDER_ID); const existing = parentFolder.getFoldersByName(primeirosNomes); const folder = existing.hasNext() ? existing.next() : parentFolder.createFolder(primeirosNomes); pastaGrupo = folder.getUrl(); } } nomes.forEach(nome => { const clienteExistente = buscarClientePorNome(nome); if (clienteExistente) { // Já existe: usa pasta do grupo ou a dele if (!pastaGrupo) pastaGrupo = clienteExistente.pastaDrive; } else { // Novo cliente const result = adicionarCliente({ nomeCompleto: nome, grupo: grupo || '', status: 'Em andamento', pastaDrive: pastaGrupo || '' }); if (result.success && !pastaGrupo) { pastaGrupo = result.pastaUrl; } } }); // Salvar arquivos UMA vez na pasta do grupo/cliente if (pastaGrupo) { if (attachments.length > 0) salvarAnexosNaPasta(pastaGrupo, attachments); if (linksDrive.length > 0) salvarLinksDrivePasta(pastaGrupo, linksDrive); } msg.markRead(); } }); }); } function extrairNomes(body) { if (!body || !body.trim()) return []; const match = body.match(/###\s*([\s\S]*?)\s*###/); if (!match) return []; const bloco = match[1]; let nomes = bloco.includes(';') ? bloco.split(';') : bloco.split('\n'); return nomes.map(n => n.trim()).filter(n => n.length > 2 && n.length < 100); } function buscarClientePorNome(nome) { const sheet = SpreadsheetApp.openById(CONFIG.SPREADSHEET_ID).getSheetByName(CONFIG.SHEET_CLIENTES); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][1].toLowerCase().trim() === nome.toLowerCase().trim()) { return { id: data[i][0], nomeCompleto: data[i][1], pastaDrive: data[i][10] }; } } return null; } function salvarAnexosNaPasta(pastaUrl, attachments) { const folderId = pastaUrl.match(/folders\/([a-zA-Z0-9_-]+)/)?.[1]; if (!folderId) return; const folder = DriveApp.getFolderById(folderId); attachments.forEach(att => { const nome = att.getName(); const existentes = folder.getFilesByName(nome); if (!existentes.hasNext()) { folder.createFile(att); } }); } // Extrai todos os anexos de uma mensagem, incluindo inline e encaminhados function obterAnexos(msg) { const todos = msg.getAttachments({ includeInlineImages: true, includeAttachments: true }); return todos.filter(att => { const name = att.getName().toLowerCase(); if (name.endsWith('.eml')) return false; return att.getSize() > 500; }); } // Extrai links do Google Drive do corpo do email function extrairLinksDrive(body) { if (!body) return []; const regex = /https:\/\/drive\.google\.com\/file\/d\/[a-zA-Z0-9_-]+/g; const matches = body.match(regex) || []; return [...new Set(matches)]; // remove duplicados } // Salva links do Drive como atalhos na pasta do cliente function salvarLinksDrivePasta(pastaUrl, links) { const folderId = pastaUrl.match(/folders\/([a-zA-Z0-9_-]+)/)?.[1]; if (!folderId || links.length === 0) return; const folder = DriveApp.getFolderById(folderId); links.forEach(link => { const fileId = link.match(/\/d\/([a-zA-Z0-9_-]+)/)?.[1]; if (!fileId) return; try { const file = DriveApp.getFileById(fileId); // Cria uma cópia na pasta do cliente file.makeCopy(file.getName(), folder); } catch (e) { // Se não tiver permissão para copiar, salva o link como arquivo de texto folder.createFile(fileId + '.txt', 'Link: ' + link); } }); } // ============================================ // CONFIGURAR TRIGGER (rodar manualmente uma vez) // ============================================ function criarTriggerEmail() { ScriptApp.newTrigger('processarEmails') .timeBased() .everyMinutes(5) .create(); } // ============================================ // VARRER EMAILS RECENTES (rodar manualmente para testar) // Busca os últimos 30 emails (lidos ou não) e processa // os que forem de remetentes cadastrados // ============================================ function varrerEmailsRecentes() { const remetentesPermitidos = listarRemetentes(); if (remetentesPermitidos.length === 0) { Logger.log('Nenhum remetente cadastrado.'); return; } Logger.log('Remetentes permitidos: ' + remetentesPermitidos.join(', ')); const threads = GmailApp.search('in:inbox newer_than:3d', 0, 30); let processados = 0; threads.forEach(thread => { const messages = thread.getMessages(); messages.forEach(msg => { const from = msg.getFrom().match(/<(.+?)>/)?.[1] || msg.getFrom(); const fromClean = from.toLowerCase().trim(); if (!remetentesPermitidos.some(r => r.toLowerCase().trim() === fromClean)) return; const body = msg.getPlainBody(); const attachments = obterAnexos(msg); const linksDrive = extrairLinksDrive(body); const subject = msg.getSubject(); Logger.log('--- Email encontrado ---'); Logger.log('De: ' + fromClean); Logger.log('Assunto: ' + subject); Logger.log('Anexos: ' + attachments.length); Logger.log('Links Drive: ' + linksDrive.length); if (attachments.length > 0) { Logger.log('Nomes dos anexos: ' + attachments.map(a => a.getName()).join(', ')); } if (linksDrive.length > 0) { Logger.log('Links: ' + linksDrive.join(', ')); } Logger.log('Corpo: ' + (body || '').substring(0, 200)); if (attachments.length === 0 && linksDrive.length === 0) { Logger.log('Sem anexos e sem links do Drive, pulando.'); return; } const nomes = extrairNomes(body); Logger.log('Nomes extraídos: ' + JSON.stringify(nomes)); if (nomes.length === 0) { Logger.log('Nenhum nome extraído do corpo, pulando.'); return; } const grupo = nomes.length > 1 ? Utilities.getUuid().split('-')[0] : null; let pastaGrupo = null; if (nomes.length > 1) { for (const nome of nomes) { const existente = buscarClientePorNome(nome); if (existente && existente.pastaDrive) { pastaGrupo = existente.pastaDrive; Logger.log('Pasta existente do grupo: ' + pastaGrupo); break; } } if (!pastaGrupo) { const primeirosNomes = nomes.map(n => n.split(' ')[0]).join(' e '); const parentFolder = DriveApp.getFolderById(CONFIG.DRIVE_FOLDER_ID); const existing = parentFolder.getFoldersByName(primeirosNomes); const folder = existing.hasNext() ? existing.next() : parentFolder.createFolder(primeirosNomes); pastaGrupo = folder.getUrl(); Logger.log('Pasta para grupo: ' + primeirosNomes); } } nomes.forEach(nome => { const clienteExistente = buscarClientePorNome(nome); if (clienteExistente) { Logger.log('Cliente existente: ' + nome); if (!pastaGrupo) pastaGrupo = clienteExistente.pastaDrive; } else { Logger.log('Novo cliente: ' + nome); const result = adicionarCliente({ nomeCompleto: nome, grupo: grupo || '', status: 'Em andamento', pastaDrive: pastaGrupo || '' }); if (result.success) { if (!pastaGrupo) pastaGrupo = result.pastaUrl; Logger.log('Criado. Pasta: ' + (pastaGrupo || result.pastaUrl)); } else { Logger.log('Erro: ' + JSON.stringify(result)); } } }); // Salvar arquivos UMA vez na pasta if (pastaGrupo) { if (attachments.length > 0) salvarAnexosNaPasta(pastaGrupo, attachments); if (linksDrive.length > 0) salvarLinksDrivePasta(pastaGrupo, linksDrive); Logger.log('Arquivos salvos na pasta: ' + pastaGrupo); } msg.markRead(); processados++; }); }); Logger.log('=== Finalizado. Emails processados: ' + processados + ' ==='); }
